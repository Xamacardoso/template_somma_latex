% --- Fontes ---
% Usa fontspec para definir as fontes exigidas (Arial Narrow e Arial)
\usepackage{fontspec}

% Tenta carregar a Arial Narrow. Caso não encontre no sistema, usa Arial.
\IfFontExistsTF{Arial Narrow}{
    \setmainfont{Arial Narrow}[
        BoldFont={Arial Narrow Bold},
        ItalicFont={Arial Narrow Italic},
        BoldItalicFont={Arial Narrow Bold Italic}
    ]
}{
    \setmainfont{Arial}
}

% Define comandos específicos para fontes do cabeçalho
\newfontfamily\arialfont{Arial}
\newfontfamily\bodonifont{Bodoni MT}
\newfontfamily\calibrifont{Calibri}
