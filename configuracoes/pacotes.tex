% --- Fontes ---
% Usa fontspec para definir as fontes exigidas (Arial Narrow e Arial)
\usepackage{fontspec}

% Arial Narrow (Usada intensamente nos títulos em PT, EN e metadados)
\setmainfont{ArialNarrow-regular.ttf}[
    Path = fonts/,
    BoldFont = ArialNarrow-bold.ttf,
    ItalicFont = ArialNarrow-italic.ttf,
    BoldItalicFont = ArialNarrow-boldItalic.ttf
]

% Bodoni MT (Usada apenas em alguns cabeçalhos)
\newfontfamily\bodonifont{Bodoni-regular.ttf}[
    Path = fonts/,
    BoldFont = Bodoni-bold.ttf,
    ItalicFont = Bodoni-italic.ttf,
    BoldItalicFont = Bodoni-boldItalic.ttf
]

% Arial Normal (usada no cabeçalho)
\IfFontExistsTF{Arial}{
    \newfontfamily\arialfont{Arial}
}{
    \newfontfamily\arialfont{ArialNarrow-regular.ttf}[
        Path = fonts/,
        BoldFont = ArialNarrow-bold.ttf,
        ItalicFont = ArialNarrow-italic.ttf,
        BoldItalicFont = ArialNarrow-boldItalic.ttf
    ]
}

% Calibri (usada na barra do título)
\IfFontExistsTF{Calibri}{
    \newfontfamily\calibrifont{Calibri}
}{
    \def\calibrifont{\sffamily}
}
